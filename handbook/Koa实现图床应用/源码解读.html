    # koa源码解读

    # koa主流程简化

    首先梳理出koa主流程的代码

    class Emitter{
    // node 内置模块
    constructor(){
    }
}
class Koa extends Emitter{
    constructor(options){
        super();
        options = options || {};
        this.middleware = [];  // 中间件列表
        this.context = {  // 请求参数
            method: 'GET',
            url: '/url',
            body: undefined,
            set: function(key, val){
                console.log('context.set', key, val);
            },
        };
    }
    use(fn){  // 中间件添加
        this.middleware.push(fn);
        return this;
    }
    listen(){  // 开启发送请求
        const  fnMiddleware = compose(this.middleware);
        const ctx = this.context;
        const handleResponse = () => respond(ctx);
        const onerror = function(){
            console.log('onerror');
        };
        fnMiddleware(ctx).then(handleResponse).catch(onerror);
    }
}
function respond(ctx){  // 接受请求
    console.log('handleResponse');
    console.log('response.end', ctx.body);
}

    其中的重点就是listen函数中的compose函数，下面来详细看一下这个函数。

    # koa-compose解析（洋葱模型实现）

    学习koa-compose之前，我们先来看两张图：

    在koa中，所有的请求都放在中间件的第一个参数context对象中了。接下来我们看下compose函数的代码

    function compose (middleware) {
  if (!Array.isArray(middleware)) throw new TypeError('Middleware stack must be an array!')
  for (const fn of middleware) {
    if (typeof fn !== 'function') throw new TypeError('Middleware must be composed of functions!')
  }

 //  传入对象 context 返回Promise
  return function (context, next) {
    // last called middleware #
    let index = -1
    return dispatch(0)
    function dispatch (i) {
      if (i <= index) return Promise.reject(new Error('next() called multiple times'))
      index = i
      let fn = middleware[i]
        if(i === middleware.length) fn = next
        try{
            return Promise.resolve(fn(context,dispatch.bind(null,i+1)))
        }catch (err) {
            return Promise.reject(err)
        }
    }
  }
}

    通过阅读上面的代码我们发现，其实compose返回的就是像下面的结构：

    const middleware = [fn1,fn2,fn3]
const fnMiddleware = function(context){
    return Promise.resolve(
        fn1(context,function next(){
            return Promise.resolve(
                fn2(context,function next(){
                    Promise.resolve(
                        fn3(context,function next(){
                            Promise.resolve()
                        })
                    )
                })
            )
        })
    )
}
fnMiddleware(ctx).then(handleResponse).catch(onerror)

    也就是说koa-compose返回的是一个Promise，Promise中取出第一个函数（app.use中添加的中间件），传入context和第一个next函数来执行 第一个next函数里返回的也是Promise，Promise中取出第二个函数（app.use中添加的中间件），传如context和第二个next函数来执行 第二个next函数里返回的也是Promise，Promise中取出第三个函数（app.use中添加的中间件），传如context和第三个next函数来执行 第三个... 依次类推，最后一个中间件中有调用next函数，则返回Promise.resolve。如果没有，则不执行next函数。这样就把所有的中间件串联起来了，这就是我们常说的洋葱模型
